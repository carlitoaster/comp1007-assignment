        EXPORTS: isEqual(Boolean)

        ALGORITHM:
                isEqual = false
                if inObject is a Astronaut
                    TRANSFORM inObject TO Astronaut named inAstronaut
                    IF name EQUALS inAstronaut.getName() THEN
                        IF role EQUALS inAstronaut.getRole() THEN
                            IF age EQUALS inAstronaut.getAge() THEN

                                    isEqual = TRUE
